// Finalization must occur only at end or above cap
ALWAYS(transaction(recv_finalize) AND now(t) AND end(e) AND raised(r) AND cap(c) IMPLIES (t >= e OR r >= c))

// Finalization can happen only once
ALWAYS(onceFinalize(true) IMPLIES NOT transaction(recv_finalize))

// No token purchases after finalization
ALWAYS(onceFinalize(true) IMPLIES NOT transaction(recv_buyToken))
